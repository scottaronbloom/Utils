cmake_minimum_required(VERSION 3.7)
if(CMAKE_VERSION VERSION_LESS "3.7.0")
    set(CMAKE_INCLUDE_CURRENT_DIR ON)
endif()

project(gif)
SET(CMAKE_MODULE_PATH ${CMAKE_CURRENT_SOURCE_DIR}/../Modules ${CMAKE_MODULE_PATH})

find_package(Qt5 COMPONENTS Core Widgets REQUIRED)
find_package(Deploy REQUIRED)

project( gif )

AddQtIncludes()

include( ${CMAKE_CURRENT_LIST_DIR}/include.cmake )
include( ${CMAKE_CURRENT_LIST_DIR}/../Project.cmake )
add_definitions( -DBUILD_GIF_PLUGIN=1 )
add_library(gif SHARED
    ${project_SRCS} 
    ${project_H}  
    ${qtproject_SRCS} 
    ${qtproject_QRC} 
    ${qtproject_QRC_SRCS} 
    ${qtproject_UIS_H} 
    ${qtproject_MOC_SRCS} 
    ${qtproject_CPPMOC_SRCS}
    ${qtproject_CPPMOC_H} 
    ${qtproject_H} 
    ${qtproject_UIS}
    ${qtproject_QRC_SOURCES}
    ${_CMAKE_FILES}
    ${_CMAKE_MODULE_FILES}
)

target_link_libraries( gif
    Qt5::Widgets
    Qt5::Core
    SABUtils
)

set_target_properties( gif PROPERTIES FOLDER SharedLibs )

INSTALL( FILES $<TARGET_FILE_DIR:gif>/$<TARGET_FILE_BASE_NAME:gif>.dll DESTINATION imageformats )
INSTALL( FILES $<TARGET_FILE_DIR:gif>/$<TARGET_FILE_BASE_NAME:gif>.pdb DESTINATION imageformats CONFIGURATIONS Debug RelWithDebInfo )

SET( _SL_FILES
    "$<TARGET_FILE:gif>"
    "$<TARGET_FILE_DIR:gif>/$<TARGET_FILE_BASE_NAME:gif>.pdb"
    )

InstallFilesPostBuild( TARGET gif INFILES ${_SL_FILES} TARGET_DIR "$<TARGET_FILE_DIR:MediaManager>" )
